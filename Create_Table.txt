CREATE TABLE USER_(
      USER_ID INT AUTO_INCREMENT PRIMARY KEY,
      FIRST_NAME VARCHAR(250),
      LAST_NAME VARCHAR(250),
      EMAIL VARCHAR(250),
      PH_NO BIGINT,
      GENDER VARCHAR(250),
      BALANCE BIGINT
);

CREATE TABLE TRANSFERS(
      TRANSFER_ID INT AUTO_INCREMENT PRIMARY KEY,
      TRANSFER_AMOUNT BIGINT,
      DEPOSITOR INT,
      CREDITER INT,
      CONSTRAINT FOREIGN KEY(DEPOSITOR) REFERENCES USER_(USER_ID),
      CONSTRAINT FOREIGN KEY(CREDITER) REFERENCES USER_(USER_ID)
);
